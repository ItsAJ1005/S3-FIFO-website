def factorial(n):       
  if n == 0:             # Base case: if n is 0, return 1
      return 1
  else:
      result = 1
      for i in range(1, n + 1):  # Loop from 1 to n
          result *= i    # Multiply result by each value of i
      return result     # Return the final result
print(factorial(5))
